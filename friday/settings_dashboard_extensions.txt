  function debounce(fn, delay) {
    var timer = null;
    return function() {
      var ctx = this, args = arguments;
      if (timer) clearTimeout(timer);
      timer = setTimeout(function() { fn.apply(ctx, args); }, delay);
    };
  }

  function throttle(fn, delay) {
    var last = 0, timer = null;
    return function() {
      var ctx = this, args = arguments, now = Date.now();
      if (now - last >= delay) {
        last = now;
        fn.apply(ctx, args);
      } else if (!timer) {
        timer = setTimeout(function() {
          last = Date.now();
          timer = null;
          fn.apply(ctx, args);
        }, delay - (now - last));
      }
    };
  }

  function fuzzyMatch(text, query) {
    if (!query) return true;
    text = text.toLowerCase();
    query = query.toLowerCase();
    var ti = 0;
    for (var qi = 0; qi < query.length; qi++) {
      var ch = query[qi];
      if (ch === ' ') continue;
      ti = text.indexOf(ch, ti);
      if (ti === -1) return false;
      ti++;
    }
    return true;
  }

  /* -------- sort -------- */
  function sortKeys(method) {
    if (method === 'default') { sortMethod = 'default'; return; }
    sortMethod = method;
    var parts = method.split('-');
    var field = parts[0];
    var dir = parts.length > 1 ? parts[1] : 'asc';
    kData.sort(function(a, b) {
      var va, vb;
      switch (field) {
        case 'name':
          va = (a.name || '').toLowerCase();
          vb = (b.name || '').toLowerCase();
          break;
        case 'category':
          va = (a.category || '').toLowerCase();
          vb = (b.category || '').toLowerCase();
          break;
        case 'status':
          va = a.present ? 1 : 0;
          vb = b.present ? 1 : 0;
          break;
        case 'pricing':
          va = (a.pricing || '').toLowerCase();
          vb = (b.pricing || '').toLowerCase();
          break;
        default: return 0;
      }
      if (va < vb) return dir === 'asc' ? -1 : 1;
      if (va > vb) return dir === 'asc' ? 1 : -1;
      return 0;
    });
  }

  function resetSort() {
    sortMethod = 'default';
    $('sortSelect').value = 'default';
  }

  function sortChangeHandler() {
    var val = $('sortSelect').value;
    if (val === 'default') {
      sortMethod = 'default';
      loadK();
    } else {
      sortKeys(val);
      rK();
    }
  }

  /* -------- export -------- */
  function exportData(format) {
    var configured = [];
    for (var i = 0; i < kData.length; i++) {
      if (kData[i].present) configured.push(kData[i]);
    }
    if (configured.length === 0) {
      t('No configured keys to export.', 'error');
      return;
    }
    var filename = 'friday-api-keys.' + (format === 'csv' ? 'csv' : 'json');
    var content = '';
    if (format === 'csv') {
      var rows = [['key','name','category','value']];
      for (var j = 0; j < configured.length; j++) {
        var k = configured[j];
        rows.push([k.key, k.name || '', k.category || '', k.value || '']);
      }
      content = rows.map(function(r) {
        return r.map(function(v) { return '"' + String(v).replace(/"/g, '""') + '"'; }).join(',');
      }).join('\n');
      content = '\uFEFF' + content;
    } else {
      var obj = {};
      for (var jj = 0; jj < configured.length; jj++) {
        obj[configured[jj].key] = configured[jj].value || '';
      }
      content = JSON.stringify(obj, null, 2);
    }
    var blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
    var url = URL.createObjectURL(blob);
    var a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    t('Exported ' + configured.length + ' keys as ' + format.toUpperCase(), 'success');
  }

  /* -------- import -------- */
  function importFromClipboard(text) {
    if (!text || !text.trim()) {
      t('No text to import.', 'error');
      return 0;
    }
    var lines = text.split('\n');
    var imported = 0;
    var matched = [];
    for (var i = 0; i < lines.length; i++) {
      var line = lines[i].trim();
      if (!line || line.startsWith('#') || line.startsWith(';')) continue;
      var eqIdx = line.indexOf('=');
      if (eqIdx === -1) continue;
      var envKey = line.substring(0, eqIdx).trim();
      var envVal = line.substring(eqIdx + 1).trim();
      if (!envKey || !envVal) continue;
      for (var j = 0; j < kData.length; j++) {
        var nameUpper = kData[j].name.toUpperCase().replace(/[^A-Z0-9]/g, '_');
        var keyUpper = kData[j].key.toUpperCase();
        if (envKey === nameUpper || envKey === keyUpper || envKey.indexOf(keyUpper) !== -1) {
          if (!kData[j].present || kData[j].value !== envVal) {
            var inp = document.querySelector('.kIn[data-key="' + kData[j].key + '"]');
            if (inp) inp.value = envVal;
            matched.push(kData[j].name);
            imported++;
          }
          break;
        }
      }
    }
    if (imported > 0) {
      t('Imported ' + imported + ' key(s): ' + matched.join(', '), 'success');
      $('dzResult').textContent = 'Imported ' + imported + ' key(s) from file.';
      $('dzResult').className = 'dz-result show dz-ok';
    } else {
      t('No matching keys found in import.', 'info');
      $('dzResult').textContent = 'No matching keys found. Check the file format (KEY_NAME=value).';
      $('dzResult').className = 'dz-result show dz-err';
    }
    return imported;
  }

  /* -------- bulk select -------- */
  function updateBulkBar() {
    var count = 0;
    for (var key in selectedKeys) {
      if (selectedKeys.hasOwnProperty(key) && selectedKeys[key]) count++;
    }
    $('bulkCount').textContent = count;
    $('bulkBar').classList.toggle('visible', count > 0);
    document.querySelectorAll('.card').forEach(function(c) {
      var k = c.dataset.key;
      c.classList.toggle('selected', !!selectedKeys[k]);
    });
  }

  function bulkSelectAll() {
    document.querySelectorAll('#kGrid .card').forEach(function(c) {
      var key = c.dataset.key;
      if (key) selectedKeys[key] = true;
    });
    updateBulkBar();
  }

  function bulkDeselectAll() {
    selectedKeys = {};
    updateBulkBar();
  }

  function getSelectedKeyList() {
    var list = [];
    for (var key in selectedKeys) {
      if (selectedKeys.hasOwnProperty(key) && selectedKeys[key]) list.push(key);
    }
    return list;
  }

  /* -------- batch verify -------- */
  function batchVerify() {
    var keys = getSelectedKeyList();
    if (keys.length === 0) {
      var cards = document.querySelectorAll('#kGrid .card');
      keys = [];
      cards.forEach(function(c) { if (c.dataset.key) keys.push(c.dataset.key); });
    }
    if (keys.length === 0) {
      t('No keys to verify.', 'info');
      return;
    }
    verifyQueue = keys.slice();
    verifyRunning = false;
    processVerifyQueue();
  }

  function processVerifyQueue() {
    if (verifyQueue.length === 0) {
      verifyRunning = false;
      t('Batch verification complete.', 'success');
      return;
    }
    if (verifyRunning) return;
    var key = verifyQueue.shift();
    var btn = document.querySelector('.tBtn[data-key="' + key + '"]');
    if (btn && !btn.disabled) {
      verifyRunning = true;
      btn.click();
      var checkDone = setInterval(function() {
        if (btn.disabled) return;
        clearInterval(checkDone);
        verifyRunning = false;
        processVerifyQueue();
      }, 200);
    } else {
      processVerifyQueue();
    }
  }

  function initKeyboardShortcuts() {
    document.addEventListener('keydown', function(e) {
      var ctrl = e.ctrlKey || e.metaKey;
      var tag = e.target.tagName;
      if (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT') {
        if (e.key === 'Escape') {
          e.target.blur();
          if ($('mOverlay').classList.contains('open')) $('mOverlay').classList.remove('open');
        }
        return;
      }
      if (e.key === '?') {
        e.preventDefault();
        $('kbdModal').classList.contains('open') ? hideKeyboardShortcuts() : showKeyboardShortcuts();
        return;
      }
      if (e.key === 'Escape') {
        if ($('kbdModal').classList.contains('open')) { hideKeyboardShortcuts(); e.preventDefault(); return; }
        if ($('exportModal').classList.contains('open')) { $('exportModal').classList.remove('open'); e.preventDefault(); return; }
        if ($('confirmOverlay').classList.contains('open')) { $('confirmCancel').click(); e.preventDefault(); return; }
        return;
      }
      if (ctrl && (e.key === 'f' || e.key === 'F')) {
        e.preventDefault();
        var activePage = document.querySelector('.page.active');
        if (activePage) {
          var search = activePage.querySelector('.search-input');
          if (search) search.focus();
        }
        return;
      }
      if (e.key === 's' || e.key === 'S') {
        if (!ctrl) {
          e.preventDefault();
          $('sortSelect').focus();
          if (!$('sortBar').classList.contains('visible')) {
            $('sortBar').classList.add('visible');
            sortVisible = true;
          }
          return;
        }
      }
      if (e.key === 'e' || e.key === 'E') {
        if (!ctrl) {
          e.preventDefault();
          $('exportModal').classList.add('open');
          return;
        }
      }
      if (e.key === 'r' || e.key === 'R') {
        if (!ctrl) {
          e.preventDefault();
          loadK();
          t('Refreshed data.', 'info');
          return;
        }
      }
      if (e.key === 'a' || e.key === 'A') {
        if (!ctrl) {
          e.preventDefault();
          bulkSelectAll();
          return;
        }
      }
      if (e.key === 'd' || e.key === 'D') {
        if (!ctrl) {
          e.preventDefault();
          bulkDeselectAll();
          return;
        }
      }
    });
  }

  /* -------- search highlight -------- */
  function searchHighlight(text) {
    var q = (text || '').trim().toLowerCase();
    document.querySelectorAll('.card-desc').forEach(function(el) {
      var orig = el.getAttribute('data-orig') || el.textContent;
      if (!el.getAttribute('data-orig')) el.setAttribute('data-orig', orig);
      if (!q || q.length < 1) {
        el.innerHTML = el.getAttribute('data-orig');
        return;
      }
      var lower = orig.toLowerCase();
      var idx = lower.indexOf(q);
      if (idx === -1) {
        el.innerHTML = orig;
        return;
      }
      var before = orig.substring(0, idx);
      var match = orig.substring(idx, idx + q.length);
      var after = orig.substring(idx + q.length);
      el.innerHTML = before + '<span class="search-highlight">' + match + '</span>' + after;
    });
  }

  /* -------- format pricing -------- */
  function formatPricing(pricing) {
    if (!pricing) return '<span class="status-badge off" style="font-size:10px;padding:1px 8px">Unknown</span>';
    var p = pricing.toLowerCase();
    if (p.indexOf('free') !== -1) return '<span class="status-badge on" style="font-size:10px;padding:1px 8px">Free</span>';
    if (p.indexOf('paid') !== -1 || p.indexOf('tier') !== -1) return '<span class="status-badge off" style="font-size:10px;padding:1px 8px;background:rgba(255,214,0,0.1);color:var(--yellow)">Paid</span>';
    return '<span class="status-badge off" style="font-size:10px;padding:1px 8px">' + pricing + '</span>';
  }

  /* -------- drag-drop -------- */
  function initDragDrop() {
    var zone = $('dropZone');
    var input = $('dropInput');
    zone.onclick = function() { input.click(); };
    zone.ondragover = function(e) { e.preventDefault(); e.stopPropagation(); zone.classList.add('drag-over'); };
    zone.ondragleave = function(e) { e.preventDefault(); e.stopPropagation(); zone.classList.remove('drag-over'); };
    zone.ondrop = function(e) {
      e.preventDefault();
      e.stopPropagation();
      zone.classList.remove('drag-over');
      var files = e.dataTransfer.files;
      if (files.length > 0) processImportFile(files[0]);
    };
    input.onchange = function() {
      if (input.files.length > 0) processImportFile(input.files[0]);
      input.value = '';
    };
  }

  function processImportFile(file) {
    if (!file) return;
    var reader = new FileReader();
    $('dzResult').className = 'dz-result show';
    $('dzResult').textContent = 'Reading ' + file.name + '...';
    reader.onload = function(e) {
      importFromClipboard(e.target.result);
    };
    reader.onerror = function() {
      $('dzResult').textContent = 'Error reading file.';
      $('dzResult').className = 'dz-result show dz-err';
    };
    reader.readAsText(file);
  }

  /* -------- scroll to top -------- */
  function initScrollToTop() {
    var btn = $('scrollTop');
    var threshold = 400;
    var onScroll = throttle(function() {
      var scrollY = window.pageYOffset || document.documentElement.scrollTop;
      btn.classList.toggle('visible', scrollY > threshold);
    }, 150);
    window.addEventListener('scroll', onScroll, { passive: true });
    btn.onclick = function() {
      window.scrollTo({ top: 0, behavior: 'smooth' });
    };
  }

    });
  }
